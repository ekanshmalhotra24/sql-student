/*
Avant l'exo 2.6.1

Bonus 1) Quelle est la moyenne de salaire des professeurs par section ? 
On veut le nom de section, id de section ainsi que le salaire moyen.
*/

SELECT s.section_name,s.section_id,AVG(p.professor_wage) as 'Salaire Moyen'
FROM section as s LEFT JOIN professor as p
	ON s.section_id=p.section_id
GROUP BY s.section_name,s.section_id



/*
Bonus 2) Par nom et prénom de professeur, quel est le maximum, 
le minimum, la moyenne et le total des ECTS des cours qu'ils donnent ?
*/

SELECT p.professor_name, p.professor_surname,
MAX(c.course_ects) AS 'MAX ECTS',MIN(c.course_ects) AS 'MIN ECTS',
AVG(c.course_ects) AS 'AVG ECTS',SUM(c.course_ects) AS 'Total ECTS'
FROM professor as p LEFT JOIN course as c
	ON p.professor_id=c.professor_id
GROUP BY p.professor_name, p.professor_surname

--Solution 1
SELECT se.section_id,del.first_name,del.last_name,AVG(DATEPART(yy,GETDATE())-year(st.birth_date)) as 'Age Moyen', (DATEPART(yy,GETDATE())-year(del.birth_date)) as 'Age Delegate'

	LEFT JOIN student as del ON del.student_id=se.delegate_id
GROUP BY se.section_id,del.last_name,del.first_name,del.birth_date
HAVING AVG(DATEPART(yy,GETDATE())-year(st.birth_date)) <((DATEPART(yy,GETDATE())-year(del.birth_date)))

--Solution 2
SELECT 
	AVG( YEAR(st.birth_date)) as moyenne_annee_date_naissance
	,se.section_id
	,info_delegate.first_name
	,info_delegate.last_name
	,YEAR(info_delegate.birth_date) as 'année date naissance délégué'
FROM student st
	LEFT JOIN section se ON se.section_id = st.section_id
	LEFT JOIN student info_delegate ON info_delegate.student_id = se.delegate_id
GROUP BY 
	se.section_id
	,info_delegate.birth_date
	,info_delegate.first_name
	,info_delegate.last_name
HAVING AVG( YEAR(st.birth_date)) > YEAR(info_delegate.birth_date)

/*
Exercice 2.7.1 – Donner la liste des étudiants (nom et prénom) qui font partie de la même 
section que mademoiselle « Roberts ». La liste doit être classée par ordre alphabétique sur le nom 
et mademoiselle « Roberts » ne doit pas apparaître dans la liste 
*/

SELECT last_name, first_name,section_id
FROM student 
WHERE section_id = (SELECT section_id FROM student
						WHERE last_name = 'Roberts') AND last_name NOT IN ('Roberts')


/*
Exercice 2.7.2 – Donner la liste des étudiants (nom, prénom et résultat) de l’ensemble des 
étudiants ayant obtenu un résultat annuel supérieur au double du résultat moyen pour l’ensemble 
des étudiants  
*/

SELECT last_name,first_name,year_result
FROM student
WHERE year_result>(SELECT AVG(year_result)*2
					FROM student				)

					WHERE section_id IS NOT NULL				)

/*
Exercice 2.7.4 – Donner la liste des étudiants qui ont comme mois de naissance le mois 
correspondant à la date d’engagement du professeur « Giot ». Classer les étudiants par ordre de 
résultat annuel décroissant   
*/

SELECT last_name,first_name,CONVERT(VARCHAR,birth_date,101) as 'Date de Naisaance',year_result
FROM student
WHERE month(birth_date)=(SELECT month(professor_hire_date) FROM professor
							WHERE professor_name='Giot')
ORDER BY year_result DESC


/*
Exercice 2.7.5 – Donner la liste des étudiants qui ont obtenu le grade « TB » pour leur résultat 
annuel  
*/

SELECT last_name,first_name,year_result
FROM student
WHERE year_result BETWEEN (
    SELECT lower_bound FROM grade WHERE grade = 'TB'
) AND (
    SELECT upper_bound FROM grade WHERE grade = 'TB'
)

Exercice 2.7.7 – Donner la liste des sections qui se composent de plus de quatre étudiants

SELECT section_id,section_name
FROM section
WHERE section_id IN (SELECT section_id FROM student
					GROUP BY section_id
